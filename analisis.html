edades = [0, 18, 30, 45, 60,100]
freq = np.zeros(len(edades) - 1)
pos = 1
for idx in range(len(por_edad)):
    if pos > len(edades) - 2:
        break
    if idx >= edades[pos]:
        pos += 1
    freq[pos - 1] += por_edad.iloc[idx]
